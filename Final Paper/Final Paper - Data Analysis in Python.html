from uszipcode import SearchEngine

def retzip(lat, lng, rad):
    search = SearchEngine()
    result = search.by_coordinates(lat, lng, radius=rad, returns=30)
    
    ls_zip = []
    for zip in result:
        ls_zip.append(zip.values()[0])
        
    return ls_zip
